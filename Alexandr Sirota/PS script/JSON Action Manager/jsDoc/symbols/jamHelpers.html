<static> {Array} jamHelpers.fromBlendRangeList(blendRangeList, explicit)

var blendRangeList =
{
    "<list>":
    [
        {
            "<object>":
            {
                "blendRange":
                {
                    "channel":
                    {
                        "<reference>": [ { "channel": { "<enumerated>": { "channel": "gray" } } } ]
